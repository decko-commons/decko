def replace_name_reference old_name, new_name
  @referee_card = nil
  @referee_name = nil
  if name.is_a? Card::Content
    name.find_chunks(Chunk::Reference).each do |chunk|
      chunk.replace_reference old_name, new_name
    end
  else
    @name = name.to_name.swap old_name, new_name
  end
end
