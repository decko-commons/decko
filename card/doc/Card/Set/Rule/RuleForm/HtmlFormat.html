view :rule_form, cache: :never, unknown: true do
  @success_view ||= :open
  @rule_context ||= card
  @form_type ||= :overlay

  wrap do
    edit_rule_form @success_view do
      [
        hidden_tags(success: @edit_rule_success),
        haml(:rule_form)
      ].join
    end
  end
end
