def undo_button
  link_to "undo", method: :post, rel: "nofollow", class: "btn btn-secondary ml-2 btn-sm btn-reduced-padding slotter",
                  remote: true,
                  "data-slot-selector": ".card-slot.quick_edit-view",

                          revert_to: :previous }
end
