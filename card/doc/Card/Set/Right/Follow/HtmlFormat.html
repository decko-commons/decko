def follow_item condition, button=true
  condition ||= "*always"
  wrap do
    card_form action: :update, success: { view: :follow_item } do
      [
        follow_item_hidden_tags(condition),
        (follow_item_button(condition) if button),
        follow_item_link(condition)
      ].compact
    end
  end
end
