def tabs tab_hash, active_name=nil, args={}, &block
  klass = args[:load] == :lazy ? Card::LazyTab : Card::Tab
  args.reverse_merge!(
    panel_attr: {},
    pane_attr: {},
    tab_type: "tabs",
    block: block,
    tab_objects: Card::Tab.tab_objects(self, tab_hash, active_name, klass)
  )
  haml :tab_panel, args
end
