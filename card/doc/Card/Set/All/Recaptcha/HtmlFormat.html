def recaptcha_token action
  output [
    javascript_include_tag(recaptcha_script_url),
    hidden_field_tag("recaptcha_token", "",
                     "data-site-key": Card.config.recaptcha_site_key,
                     "data-action": action,
                     class: "_recaptcha-token")
  ]
end
