def wrap_help_text text
  help = text
  if (rule_card = card.help_rule_card)
    edit_link = with_nest_mode(:normal) { nest(rule_card, view: :edit_link) }
    help = "<span class='d-none'>#{edit_link}</span>#{text}"
  end
  help
end
