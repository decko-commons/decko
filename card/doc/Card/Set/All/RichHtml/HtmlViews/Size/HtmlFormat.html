view :max_size do
  if voo.size.is_a?(String) && voo.size.match(/^\d+x\d+\$/)
    max_size(*voo.size.split("x"))
  else
    px = SIZE_IN_PX[voo.size&.to_sym] || 200
    max_size px, px
  end
end
