def absolutize_mark
  return unless mark.name? && (supercard = opts.dig(:new, :supercard))
  @mark = mark.absolute_name supercard.name
end

def normalize_args args
  @opts = args.last.is_a?(Hash) ? args.pop : {}
  @mark = Card.id_or_name args
end
