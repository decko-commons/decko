  exit $CHILD_STATUS.exitstatus unless system command
end

def split_args args
  before_split = true
  decko_args, command_args =
    args.partition do |a|
      before_split = (a == "--" ? false : before_split)
    end
  command_args.shift
  [decko_args, command_args]
end
