def initialize rake_task, args={}
  @task = rake_task
  opts = {}
  if args.is_a? Array
    Parser.new(rake_task, opts).parse!(args)
  else
    opts = args
  end
  @envs = Array(opts[:envs])
end

def commands
  task_cmd = "bundle exec rake #{@task}"
  return [task_cmd] if !@envs || @envs.empty?
  @envs.map do |env|
    "env RAILS_ENV=#{env} #{task_cmd}"
  end
end

def run

    puts cmd
