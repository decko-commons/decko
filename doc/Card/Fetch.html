def normalize_args args
  @opts = args.last.is_a?(Hash) ? args.pop : {}
  @mark = Card.id_or_name args
  normalize_mark
end

def normalize_mark
  return unless mark.name?

  supercard = opts.dig :new, :supercard
  @mark = mark.absolute_name supercard.name if supercard
end
