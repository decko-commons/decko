def initialize opts
  opts ||= {}
  @remaining_chars = opts[:length] || 50
  @joint = opts[:joint] || "..."

  @summary = nil
  @chunks = []
  @content_omitted = false
end

def rendered
  @summary ||=
    begin
      truncate_overlap
      @chunks.map do |chunk|
        @content_omitted ||= chunk[:action] == :ellipsis
        render_chunk chunk[:action], chunk[:text]
      end.join
    end
end
