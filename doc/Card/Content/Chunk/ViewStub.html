def interpret_format_opts
  hash = @stub_hash[:format_opts]
  hash.symbolize_keys!
  hash[:nest_mode] = hash[:nest_mode].to_sym
  hash[:override] = hash[:override] == "true"
  hash[:context_names].map!(&:to_name)
end
