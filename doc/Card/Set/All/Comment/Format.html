def comment_author
  if Auth.signed_in?
    "[[#{Auth.current.name}]]"
  else
    "#{card.comment_author} (Not signed in)"
  end
end

  wrap_with :div, class: "comment-box nodblclick" do

      [hidden_comment_fields, comment_box, comment_buttons]
    end
  end
end
