def navbar id, opts={}, &block
  nav_opts = opts[:navbar_opts] || {}
  nav_opts[:class] ||= opts[:class]
  add_class nav_opts,
            "navbar navbar-dark bg-#{opts.delete(:navbar_type) || 'primary'}"
  navbar_content id, opts, nav_opts, &block
end
