def each_valid_item

    @item = @items.shift
    yield unless invalid_item?
  end
end

def pointer_item
  @items.insert 0, @item.item_cards.reject(&:unknown?)
  @items.flatten!
  record_item
  @item if @item != @machine_card && @item.known?
end
