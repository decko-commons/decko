def compile_scss scss, style=:expanded
  SassC::Engine.new(scss, style: style).render
rescue SassC::SyntaxError => e
  raise Card::Error,
        "SassC::SyntaxError (#{card.name}:#{e.sass_backtrace}): #{e.message}"
