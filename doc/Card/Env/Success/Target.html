def card name_context=@name_context
  if @card
    @card
  elsif @id
    Card.fetch @id
  elsif @name
    Card.fetch @name.to_name.absolute(name_context), new: @new_args
  end
end

def mark= value
  case value
  when Integer then @id = value
  when Card then @card = value
  else
    self.target = value
  end
end
