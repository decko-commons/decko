def action_summary type
  count = count_types[type]
  return unless count.positive?

  "#{@format.action_icon type}<small> #{count if count > 1}</small>"
end

def count_types
  @count_types ||=
    approved_actions.each_with_object(
      Hash.new { |h, k| h[k] = 0 }
    ) do |action, type_cnt|
      type_cnt[action.action_type] += 1
    end
end
