def dump
  dump_tables.each do |table|
    i = "000" # TODO: use card keys instead (this is just a label)
    write_seed_file table do
      yamlize_records table do |record, hash|
        hash["#{table}_#{i.succ!}"] = record
      end
    end
  end
end
