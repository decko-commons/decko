end

def include? subname
  subkey = subname.to_name.key
  key =~ /(^|#{JOINT_RE})#{Regexp.quote subkey}($|#{JOINT_RE})/
end

def valid?
  return true if self.class.nothing_banned?

  !parts.find do |pt|
    pt.match self.class.banned_re
  end
end
