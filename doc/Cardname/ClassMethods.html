def new obj
  return obj if obj.is_a? self.class

  str = stringify(obj)
  cache[str] ||= super(str)
end

def nothing_banned?
  return @nothing_banned unless @nothing_banned.nil?

  @nothing_banned = banned_array.empty?
end
