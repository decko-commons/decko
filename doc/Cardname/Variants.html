def simple_key
  return "" if empty?

  decoded
    .underscore
    .gsub(/[^#{OK4KEY_RE}]+/, "_")
    .split(/_+/)
    .reject(&:empty?)
    .map { |key| stable_key(key) }
    .join("_")
end

def url_key
  part_names.map do |part_name|
    stripped = part_name.decoded.gsub(/[^#{OK4KEY_RE}]+/, " ").strip
    stripped.gsub(/[\s_]+/, "_")
  end * self.class.joint
end
